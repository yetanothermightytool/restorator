// Fetch Veeam CVEs from NVD API v2 async function fetchCVEs(daysBack = 90) { const end = new Date().toISOString(); const start = new Date( Date.now() - daysBack * 86_400_000 ).toISOString(); const url = new URL( 'https://services.nvd.nist.gov' +'/rest/json/cves/2.0' ); url.searchParams.set('cpeName', 'cpe:2.3:a:veeam:backup_and_replication:*'); url.searchParams.set('pubStartDate', start); url.searchParams.set('pubEndDate', end); const { vulnerabilities } = await ( await fetch(url, { headers: { 'apiKey': NVD_API_KEY } }) ).json(); return vulnerabilities.map(({ cve }) => ({ id: cve.id, cvss: cve.metrics?.cvssMetricV31 ?.[0]?.cvssData?.baseScore, desc: cve.descriptions .find(d => d.lang === 'en')?.value })); }
